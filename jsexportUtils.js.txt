// exportUtils.js
// Handles Excel export using SheetJS with clean formatting.

export class ExcelExporter {
  /**
   * Exports structured data to an Excel file.
   * @param {Array<Object>} data - Array of objects representing rows
   * @param {string[]} headers - Column headers
   * @param {string} fileName - Output file name
   */
  exportToExcel(data, headers, fileName = 'Extracted_Data.xlsx') {
    // Create workbook and worksheet
    const wb = XLSX.utils.book_new();
    
    // Transform data into array of arrays for SheetJS
    const rows = data.map(row => {
      return headers.map(header => row[header] || '');
    });
    
    // Prepend headers
    const sheetData = [headers, ...rows];
    const ws = XLSX.utils.aoa_to_sheet(sheetData);
    
    // Auto column width calculation
    const colWidths = headers.map((header, i) => {
      let maxLen = header.length;
      rows.forEach(row => {
        const cellValue = String(row[i] || '');
        if (cellValue.length > maxLen) maxLen = cellValue.length;
      });
      return { wch: Math.min(maxLen + 5, 50) }; // Cap width at 50
    });
    ws['!cols'] = colWidths;
    
    // Add worksheet to workbook
    XLSX.utils.book_append_sheet(wb, ws, 'Extracted Data');
    
    // Trigger download
    XLSX.writeFile(wb, fileName);
  }

  /**
   * Generates a sample Excel template based on current headers.
   */
  downloadSampleTemplate(headers) {
    const sampleData = [headers.reduce((acc, h) => ({ ...acc, [h]: 'Sample Value' }), {})];
    this.exportToExcel(sampleData, headers, 'Sample_Template.xlsx');
  }
}
